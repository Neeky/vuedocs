## 文件插值
   ****
   ```html
   <html>
       <head>
           <title>直接引入vue的js文件</title>
           <!-- 开发版没有经过压缩 -->
           <!-- https://vuejs.org/js/vue.js-->
           <script src="https://cdn.jsdelivr.net/npm/vue"></script>
           <meta charset="utf8">
       </head>
   
       <body>
           <div id="app">
               {{ message }}
           </div>
       </body>
   
       <script>
            // 这个变量名可以与 id 名一致也可以不一致，没有半毛钱关系
            var app = new Vue(
               {
                   el:'#app', // 元素 id 前面一定要加 “#” 号，这个应该是类 jqury 语法
                   data:{
                       message:'hello world'
                   }
               }
           )
       </script>
   </html>
   ```

## v-if条件
   ```html
   <html>
       <head>
           <title>直接引入vue的js文件</title>
           <!-- 开发版没有经过压缩 -->
           <!-- https://vuejs.org/js/vue.js-->
           <script src="https://cdn.jsdelivr.net/npm/vue"></script>
           <meta charset="utf8">
       </head>
   
       <body>
           <div id="app-3" v-if="seen">
               是否可以这个要视条件而定
           </div>
       </body>
   
       <script>
            var app = new Vue(
               {
                   el:'#app-3',
                   data:{
                       message:'hello world',
                       seen:true
                   }
               }
           )
           // 在控制台下执行 app.seen=false 元素app-3就变得不可见了
       </script>
   
   </html>
   ```

## v-on事件监听
   ****
   ```html
   <html>
       <head>
           <title>直接引入vue的js文件</title>
           <!-- 开发版没有经过压缩 -->
           <!-- https://vuejs.org/js/vue.js-->
           <script src="https://cdn.jsdelivr.net/npm/vue"></script>
           <meta charset="utf8">
       </head>
   
       <body>
           <div id="app-5">
               <p>{{ message }}</p>
               <button v-on:click="reverseMessage">点我</button>
           </div>
       </body>
   
       <script>
            var app = new Vue(
               {
                   el:'#app-5',
                   data:{
                       message:'123456'
                   },
                   methods:{
                       reverseMessage:function(){
                           this.message = this.message.split('').reverse().join('')
                           // 我们更新了应用的状态，但没有触碰 DOM——所有的 DOM 操作都由 Vue 来处理，你编写的代码只需要关注逻辑层面即可。
                       }
                   }
               }
           )
       </script>
   
   </html>
   ```
